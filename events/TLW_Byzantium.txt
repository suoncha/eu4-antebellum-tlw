				value = 1
			}
		}

	after = {
		hidden_effect = {
			if = {
				limit = {
					mission_completed = byzantium_iron
				}
				clr_country_flag = had_16th_crisis

					has_country_flag = byz_restore_republic

					}
				}

					faction = faction_byzlatin
					influence = 100
				}
				add_faction_influence = {
					faction = faction_byzlatin
					influence = -40
				}
			}
			else_if = {
				limit = {
					has_country_flag = byz_court_mixed
				}
				add_faction_influence = {
