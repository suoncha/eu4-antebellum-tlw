					}
				}
			}
		}
	}

		days = 1
	}

		hidden_effect = {
			random_subject_country = {
				limit = {
					is_subject_of_type = diocese_vassal
					any_neighbor_country = {
						is_subject_of_type = diocese_vassal
						NOT = {
							has_opinion_modifier = {
								who = PREV
								modifier = opinion_dioceses_border_clash

						}
					}
				}
				save_event_target_as = flavorDiocese1

						NOT = {

						}
					}
					save_event_target_as = flavorDiocese2
