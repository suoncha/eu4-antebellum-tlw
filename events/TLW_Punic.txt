namespace = ab_punic
normal_or_historical_nations = yes

#Trade company & colony culture spread
province_event = {
	id = ab_punic.0
	title = ab_punic.0.t
	desc = ab_punic.0.d
	picture = CULTURE_eventPicture
	
	hidden = yes
		
	trigger = {
		NOT = { culture = owner }
		OR = {
			AND = {
				owner = {
					is_colonial_nation = yes
				}
				OR = {
					NOT = { nationalism = 1 } 
					culture_group = owner
				}
			}
			AND = {
				is_owned_by_trade_company = yes
				num_investments_in_trade_company_region = {
					investment = trade_company_cultural_center
					value = 1
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 0.5
			religion_group = owner
		}
		modifier = {
			factor = 0.1
			culture_group = owner
		}
	}

	option = {
		name = ab_punic.0.a
		change_culture = owner
		change_original_culture = owner
	}
}

#First event
country_event = {
	id = ab_punic.1
	title = ab_punic.1.t
	desc = ab_punic.1.d
	picture = DHOW_AT_SEA_eventPicture
	
	is_triggered_only = yes
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = punic_peace
				duration = -1
			}
			add_stability = 1
			add_adm_tech = 2
			add_dip_tech = 2
			add_mil_tech = 2
		}
	}
	
	option = {
		name = ab_punic.1.a
		custom_tooltip = PUNIC_EARLY_EVENT_CHAIN
		tooltip = {
			add_country_modifier = {
				name = punic_peace
				duration = -1
			}
		}
	}
}

#North America Setup
country_event = {
	id = ab_punic.2
	title = ab_punic.2.t
	desc = ab_punic.2.d
	picture = DHOW_AT_SEA_eventPicture
	
	is_triggered_only = yes
	
	fire_only_once = yes

	hidden = yes
	
	option = {
		name = ab_punic.2.a
		every_country = {
			limit = {
				technology_group = north_american
			}
			add_country_modifier = {
				name = punic_primitive
				duration = -1
			}
			# Punic League
			if = {
				limit = {
					culture = punic
				}
				country_event = {
					id = ab_punic.1
				}
			}
		}
	}
}
