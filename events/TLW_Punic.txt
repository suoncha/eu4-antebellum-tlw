	id = ab_punic.0
	title = ab_punic.0.t
	desc = ab_punic.0.d
	picture = CULTURE_eventPicture
	
	hidden = yes
		
	trigger = {
		owner = {
			is_colonial_nation = yes
		}
		NOT = { culture = owner }
		OR = {

		}
	}

		months = 600
		modifier = {
			factor = 0.5
			religion_group = owner
		}

			factor = 0.1
			culture_group = owner
		}
	}

	option = {
		name = ab_punic.0.a
		change_culture = owner
		change_original_culture = owner
	}
}
