province_event = {
	id = ab_punic.0
	title = ab_punic.0.t
	desc = ab_punic.0.d
	picture = CULTURE_eventPicture
	
	hidden = yes
		
	trigger = {
		NOT = { culture = owner }
		OR = {
			AND = {
				owner = {
					is_colonial_nation = yes
				}
				OR = {
					NOT = { nationalism = 1 } 
					culture_group = owner
				}
			}
			AND = {
				is_owned_by_trade_company = yes
				num_investments_in_trade_company_region = {
					investment = trade_company_cultural_center
					value = 1
				}
			}
		}
	}

	mean_time_to_happen = {
		months = 600
		modifier = {
			factor = 0.5
			religion_group = owner
		}
		modifier = {
			factor = 0.1
			culture_group = owner
		}
	}

	option = {
		name = ab_punic.0.a
		change_culture = owner
		change_original_culture = owner
	}
}
