			}

	}

	}

			factor = 3

				tooltip = BYZ_RIVAL_CHINA

			}
		}

			factor = 1
			custom_trigger_tooltip = {
				tooltip = BYZ_LESSTHAN0_CHINA

					NOT = {

							who = ROOT

			factor = 1

				tooltip = BYZ_LESSTHAN50_CHINA

			factor = 2

				tooltip = BYZ_LESSTHAN100_CHINA
				event_target:EmperorOfChina = {

						has_opinion = {
							who = ROOT
							value = -100

			}
