byzantium_persian_war = {
	potential = {
		was_tag = BYZ
		mission_completed = byzantium_manzikert
	}
	
	can_start = {
		tag = ROM
		mission_completed = byzantium_mesopotamia
		custom_trigger_tooltip = {
			tooltip = BYZ_NEIGHBOR_PERSIA
			OR = {
				is_neighbor_of = PER
				is_neighbor_of = PRC
				is_neighbor_of = PMN
				is_neighbor_of = PRZ
			}
		}
	}
	
	can_stop = {
		has_any_disaster = yes
	}
	
	progress = {
		modifier = {
			factor = 100
			tag = ROM
		} 
	}
	
	modifier = {
		global_unrest = 10
		stability_cost_modifier = 0.5
	}
	
	can_end = {
		is_at_war = no
		NOT = { num_of_rebel_controlled_provinces = 1 }
		NOT = { num_of_rebel_armies = 1 }
		stability = 1
		custom_trigger_tooltip = {
			tooltip = BYZ_RESOLVED_ANGELID
			has_country_flag = had_fall_angelid
		}
	}
	
	on_start = ab_byzantium_disaster.0
	on_end = ab_byzantium_disaster.1
}
