byzantium_roman_civil_war = {
	potential = {
		has_government_attribute = ab_has_dioceses
	}
	
	can_start = {
		has_any_disaster = no
		num_subjects_of_type_with_liberty_desire = {
			type = diadochi_vassal
			liberty_desire = 50
			value = 2
		}
		if = {
			limit = {
				has_country_flag = byz_roman_civil_war_end
			}
			custom_trigger_tooltip = {
				tooltip = BYZ_CIVIL_WAR_END_5YEARS
				had_country_flag = {
					flag = byz_roman_civil_war_end
					days = 1825
				}
			}
		}
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			NOT = {
				any_subject_country = {
					is_subject_of_type = diocese_vassal
					liberty_desire = 50
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.25
			num_subjects_of_type_with_liberty_desire = {
				type = diadochi_vassal
				liberty_desire = 50
				value = 2
			}
		}
		modifier = {
			factor = 0.25
			num_subjects_of_type_with_liberty_desire = {
				type = diadochi_vassal
				liberty_desire = 50
				value = 3
			}
		}
		modifier = {
			factor = 0.5
			num_subjects_of_type_with_liberty_desire = {
				type = diadochi_vassal
				liberty_desire = 50
				value = 4
			}
		}
		modifier = {
			factor = 1
			num_subjects_of_type_with_liberty_desire = {
				type = diadochi_vassal
				liberty_desire = 50
				value = 5
			}
		}
		modifier = {
			factor = 2
			num_subjects_of_type_with_liberty_desire = {
				type = diadochi_vassal
				liberty_desire = 50
				value = 6
			}
		}
	}
	
	modifier = {
		reduced_liberty_desire = -20
		can_not_declare_war = yes
	}
	
	can_end = {
		is_at_war = no
		custom_trigger_tooltip = {
			tooltip = BYZ_UNCONTESTED_EMPEROR
			has_country_flag = byz_uncontested_emperor
		}
	}
	
	on_start = ab_byzantium_disaster.56
	on_end = ab_byzantium_disaster.57
}
