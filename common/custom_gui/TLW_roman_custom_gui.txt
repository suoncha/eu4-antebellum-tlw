#Diocesis Russia
custom_button = {
    name = ab_diocese_russia_feed
    potential = {
        has_government_attribute = ab_has_dioceses
        has_country_flag = byz_diocese_menu_opened
        overlord_of = UKR
        UKR = { is_subject_of_type = grand_diocese_vassal }
    }
    trigger = { 
        any_owned_province = {
            OR = {
                AND = {
                    superregion = eastern_europe_superregion
                    NOT = { 
                        OR = {
                            region = balkan_region
                            region = carpathia_region
                            area = crimea_area
                        }
                    }
                }
                area = circassia_area
                area = dagestan_area
                AND = {
                    superregion = tartary_superregion
                    NOT = { region = tibet_region }
                }
                AND = {
                    region = manchuria_region
                    NOT = { area = liaoning_area }
                }
            }
        }
		is_at_war = no
		is_free_or_tributary_trigger = yes
		has_any_disaster = no
		is_nomad = no
		NOT = { overextension_percentage = 0.001 }
    }
    effect = { 
        hidden_effect = {
            eastern_europe_superregion = {
                limit = {
                    owned_by = ROOT
                    NOT = { 
                        OR = {
                            region = balkan_region
                            region = carpathia_region
                            area = crimea_area
                        }
                    }
                }
                add_core = UKR
                cede_province = UKR
            }
            circassia_area = {
                limit = {
                    owned_by = ROOT
                }
                add_core = UKR
                cede_province = UKR
            }
            dagestan_area = {
                limit = {
                    owned_by = ROOT
                }
                add_core = UKR
                cede_province = UKR
            }
            tartary_superregion = {
                limit = {
                    owned_by = ROOT
                    NOT = { region = tibet_region }
                }
                add_core = UKR
                cede_province = UKR
            }
            manchuria_region = {
                limit = {
                    owned_by = ROOT
                    NOT = { area = liaoning_area }
                }
                add_core = UKR
                cede_province = UKR
            }
        }
    }
    tooltip = AB_DIOCESE_RUSSIA_FEED
}

                add_government_reform = rom_diocese
