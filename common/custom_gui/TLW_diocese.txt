#Diocesis Graecia
custom_button = {
    name = ab_diocese_graecia_create
    potential = {
        NOT = { exists = RDC }
    }
    trigger = {
        owns_core_province = 151
        if = {
            limit = {
                has_country_flag = rom_caucasia_isof_anatolia
            }
            capital_scope = {
                NOT = { region = balkan_region }
                NOT = { region = anatolia_region }
                NOT = { region = caucasia_region }
            }
        }
        else = {
            capital_scope = {
                NOT = { region = balkan_region }
                NOT = { region = anatolia_region }
            }
        }
		is_at_war = no
		is_subject = no
		has_any_disaster = no
		NOT = { overextension_percentage = 0.001 }
    }
    effect = {
        hidden_effect = {
            if = {
                limit = {
                    has_country_flag = rom_caucasia_isof_anatolia
                }
                every_owned_province = {
                    limit = {
                        OR = {
                            region = anatolia_region
                            area = macedonia_area
                            area = northern_greece_area
                            area = morea_area
                            area = thrace_area
                            area = ionian_islands_area
                            region = caucasia_region
                        }
                        NOT = {
                            OR = {
                                area = circassia_area
                                area = dagestan_area
                                province_id = 4780
                                province_id = 3001
                                province_id = 1764
                            }
                        }
                    }
                    add_core = RDC
                    remove_core = ROOT
                }
            }
            else = {
                every_owned_province = {
                    limit = {
                        OR = {
                            region = anatolia_region
                            area = macedonia_area
                            area = northern_greece_area
                            area = morea_area
                            area = thrace_area
                            area = ionian_islands_area
                        }
                        NOT = {
                            OR = {
                                province_id = 4780
                                province_id = 3001
                                province_id = 1764
                            }
                        }
                    }
                    add_core = RDC
                    remove_core = ROOT
                }
            }
            release = RDC
            create_subject = {
                who = RDC
                subject_type = diocese_vassal
            }
            RDC = {
                change_primary_culture = greco_roman
                change_religion = ROOT
                change_government = republic
                add_government_reform = rom_diocese
                kill_ruler = yes
                set_capital = 151
                capital_scope = {
                    province_event = {
                        id = ab_byzantium.71
                    }
                }
            }
        }
        if = {
            limit = {
                has_country_flag = rom_caucasia_isof_anatolia
            }
            custom_tooltip = AB_DIOCESE_CAUCASIA
        }
    }
    tooltip = AB_DIOCESE_GRAECIA
}
