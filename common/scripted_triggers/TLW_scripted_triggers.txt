	OR = {
		tag = BYZ
		tag = SER
		tag = RUM
		AND = {
			tag = UKR
			NOT = { was_tag = NOV }
		}
		tag = LAT
		tag = BRN
		tag = VSG
		AND = {
			OR = {
				tag = POL
				tag = PLC
			}
			religion_group = christian
		}
		tag = AQT
		tag = SWA
		tag = RVN
		was_tag = RVN
		tag = CAP
		was_tag = CAP
	}
}
