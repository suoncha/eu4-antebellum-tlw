ab_game_rules_back_page_effect = {
    # First Page
    if = {
        limit = {
            ab_game_rules_page = { page = 1 }
        }
        set_variable = {
            which = ab_game_rules_page_var
            value = 4 # TLW
        }
    }
    else = {
        subtract_variable = {            
            which = ab_game_rules_page_var
            value = 1
        }
    }
}

ab_game_rules_next_page_effect = {
    # Last Page
    if = {
        limit = {
            ab_game_rules_page = { page = 4 } # TLW
        }
        set_variable = {
            which = ab_game_rules_page_var
            value = 1
        }
    }
    else = {
        change_variable = {            
            which = ab_game_rules_page_var
            value = 1
        }
    }
}

ab_game_rules_back_option_effect = {
	# First Option
	if = {
		limit = {
			is_variable_equal = {
				which = $var$
				value = 1
			}
		}
		set_variable = {
			which = $var$
			value = $num$
		}
	}
	else = {
		subtract_variable = {            
			which = $var$
			value = 1
		}
	}
}
