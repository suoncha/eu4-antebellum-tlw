ab_punic_league_calc_effect = {
	REB = {
		### Reset variables
		if = {
			limit = {
				has_global_flag = north_america_european_threat
			}
			set_variable = {
				which = punic_league_growth
				value = 0.5
			}
		}
		else = {
			set_variable = {
				which = punic_league_growth
				value = 0.1
			}
		}

		### Calc League Authority
		# Stability
		set_variable = {
			which = punic_league_stability
			value = 0
		}
		every_country = {
			limit = {
				tlw_is_punic_city_state_trigger = yes
			}
			REB = {
				export_to_variable = {
					which = punic_league_stability_member
					value = stability
					who = PREV
				}
				change_variable = {            
					which = punic_league_stability
					which = punic_league_stability_member
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					which = punic_league_stability
					value = 0.001
				}
			}
			divide_variable = {
				which = punic_league_stability
				value = 1000
			}
		}
		else = {
			divide_variable = {
				which = punic_league_stability
				value = 100
			}
		}

		# Income Balance
		set_variable = {
			which = punic_league_income_balance
			value = 0
		}
		every_country = {
			limit = {
				tlw_is_punic_city_state_trigger = yes
			}
			REB = {
				export_to_variable = {
					which = punic_league_income_balance_member
					value = trigger_value:current_income_balance
					who = PREV
				}
				change_variable = {            
					which = punic_league_income_balance
					which = punic_league_income_balance_member
				}
			}
		}
		if = {
			limit = {
				check_variable = {
					which = punic_league_income_balance
					value = 0.001
				}
			}
			divide_variable = {
				which = punic_league_income_balance
				value = 1000
			}
		}
		else = {
			divide_variable = {
				which = punic_league_income_balance
				value = 100
			}
		}

		# Province Count
		set_variable = {
			which = punic_league_province_count
			value = 0
		}
		every_country = {
			limit = {
				OR = {
					tlw_is_punic_city_state_trigger = yes
					is_subject_of_type = punic_tribe_vassal
				}
			}
			REB = {
				export_to_variable = {
					which = punic_league_province_count_member_states
					value = trigger_value:num_of_provinces_in_states
					who = PREV
				}
				export_to_variable = {
					which = punic_league_province_count_member_territories
					value = trigger_value:num_of_provinces_in_territories
					who = PREV
				}
				change_variable = {            
					which = punic_league_province_count
					which = punic_league_province_count_member_states
				}
				change_variable = {            
					which = punic_league_province_count
					which = punic_league_province_count_member_territories
				}
			}
		}
		divide_variable = {
			which = punic_league_province_count
			value = 1000
		}

		# Finalize
		change_variable = {
			which = punic_league_growth
			which = punic_league_stability
		}
		change_variable = {
			which = punic_league_growth
			which = punic_league_income_balance
		}
		change_variable = {
			which = punic_league_growth
			which = punic_league_province_count
		}
		change_variable = {
			which = punic_league_value
			which = punic_league_growth
		}
		if = {
			limit = {
				check_variable = {
					which = punic_league_value
					value = 100
				}
			}
			set_variable = {
				which = punic_league_value
				value = 100
			}
		}
		else_if = {
			limit = {
				NOT = {
					check_variable = {
						which = punic_league_value
						value = 0.001
					}
				}
			}
			set_variable = {
				which = punic_league_value
				value = 0
			}
		}
	}
	
	REB = {
		if = {
			limit = {
				NOT = { has_country_flag = punic_league_reform_passed_10 }
			}
			### Calc Election Point
			every_country = {
				limit = {
					tlw_is_punic_city_state_trigger = yes
				}
				ab_punic_league_election_point_calc_effect = yes
			}

			### Calc Time
			subtract_variable = {            
				which = punic_league_time
				value = 1
			}
			if = {
				limit = {
					is_variable_equal = {
						which = punic_league_time
						value = 0
					}
				}
				set_variable = {
					which = punic_league_time
					value = 60
				}
				event_target:easimaTarget = {
					ab_punic_league_new_easima_effect = yes
				}
			}
		}
	}
}

ab_punic_league_election_point_calc_effect = {
	### Reset variables
	set_variable = {
		which = election_point_growth
		which = election_point_base
	}

	### Calc Election Point
	# Diplomatic Reputation
	export_to_variable = {
		which = election_point_diplomatic_reputation
		value = trigger_value:diplomatic_reputation
	}
	round_variable = {
		which = election_point_diplomatic_reputation
		value = 0
	}

	# Good Relations with City States
	set_variable = {
		which = election_point_high_relations
		value = 0
	}
	every_country = {
		limit = {
			tlw_is_punic_city_state_trigger = yes
			NOT = { tag = ROOT }
			has_opinion = {
				who = ROOT
				value = 100
			}
		}
		ROOT = {
			change_variable = {            
				which = election_point_high_relations
				value = 1
			}
		}
	}

	# Diplomatic Advisor
	set_variable = {
		which = election_point_diplomatic_advisor
		value = 0
	}
	if = {
		limit = {
			has_dip_advisor_of_level = { level = 5 }
		}
		set_variable = {
			which = election_point_diplomatic_advisor
			value = 5
		}
	}
	else_if = {
		limit = {
			has_dip_advisor_of_level = { level = 4 }
		}
		set_variable = {
			which = election_point_diplomatic_advisor
			value = 4
		}
	}
	else_if = {
		limit = {
			has_dip_advisor_of_level = { level = 3 }
		}
		set_variable = {
			which = election_point_diplomatic_advisor
			value = 3
		}
	}
	else_if = {
		limit = {
			has_dip_advisor_of_level = { level = 2 }
		}
		set_variable = {
			which = election_point_diplomatic_advisor
			value = 2
		}
	}
	else_if = {
		limit = {
			has_dip_advisor_of_level = { level = 1 }
		}
		set_variable = {
			which = election_point_diplomatic_advisor
			value = 1
		}
	}

	# Finalize
	change_variable = {
		which = election_point_growth
		which = election_point_diplomatic_reputation
	}
	change_variable = {
		which = election_point_growth
		which = election_point_high_relations
	}
	change_variable = {
		which = election_point_growth
		which = election_point_diplomatic_advisor
	}
	change_variable = {
		which = election_point
		which = election_point_growth
	}
	if = {
		limit = {
			NOT = {
				check_variable = {
					which = election_point
					value = 0.001
				}
			}
		}
		set_variable = {
			which = election_point
			value = 0
		}
	}
}
