TLW_culture_conversion_calc_effect = {
	set_variable = {
		which = culture_conversion_var
		value = 0
	}
	set_variable = {
		which = culture_conversion_change_var
		value = 0
	}
	# Colony
	if = {
		limit = {
			NOT = { culture = owner }
			owner = {
				is_colonial_nation = yes
			}
		}
		province_event = {
			id = ab_culture_conversion.0
		}
	}
	# Hellenization
	if = {
		limit = {
			NOT = { culture_group = hellenic_g }
			owner = {
				culture_group = hellenic_g
				OR = {
					is_subject_of_type = diadochi_vassal
					has_government_attribute = ab_has_diadochi
				}
			}
		}
		province_event = {
			id = ab_culture_conversion.1
		}
	}
	# Romanization
	if = {
		limit = {
			NOT = { culture_group = roman_g }
			owner = {
				culture_group = roman_g
			}
			province_is_diocese_land_trigger = yes
		}
		province_event = {
			id = ab_culture_conversion.2
		}
	}
	# Spread of Greek Culture in Anatolia
	if = {
		limit = {
			region = anatolia_region
			NOT = { culture = greek }
			owner = {
				OR = {
					AND = {
						mission_completed = byzantium_frontier
						tag = BYZ
					}
					AND = {
						mission_completed = lat_avenge
						tag = LAT
					}
				}
			}
		}
		province_event = {
			id = ab_culture_conversion.3
		}
	}
	# Spread of Litvin Culture in Baltic
	if = {
		limit = {
			region = baltic_region
			NOT = { culture_group = east_slavic }
			owner = {
				has_country_flag = rus_litvin_spread
				culture_group = east_slavic
			}
		}
		province_event = {
			id = ab_culture_conversion.4
		}
	}
	# Spread of Visigothic Culture
	if = {
		limit = {
			owner = {
				primary_culture = Visigothic
			}
			NOT = { culture = Visigothic }
			OR = {
				AND = {
					OR = {
						area = pyrenees_area
						area = guyenne_area
						area = languedoc_area
						area = savoy_dauphine_area
						area = massif_central_area
						area = poitou_area
						area = provence_area
					}
					owner = {
						mission_completed = vsg_visigothic_aquitania
					}
				}
				AND = {
					region = iberia_region
					owner = {
						mission_completed = vsg_finish_reconquista
					}
				}
			}
		}
		province_event = {
			id = ab_culture_conversion.5
		}
	}
}
